LATAR BELAKANG

Kepadatan kendaraan dan kurangnya sistem parkir terintegrasi menjadi masalah utama di area perkotaan. Smart Parking Management hadir sebagai solusi cerdas menggunakan pendekatan Object Oriented Programming (OOP) untuk memodelkan kendaraan, slot parkir, transaksi, dan petugas secara real-time. Dengan OOP, sistem ini mudah dikembangkan, mendukung berbagai tipe kendaraan, dan memberikan transparansi ketersediaan slot.

Mengapa OOP? & Objek Smart Parking

Pendekatan OOP memetakan entitas dunia nyata (kendaraan, slot, transaksi) menjadi objek yang memiliki data dan perilaku masing-masing. Hal ini mengurangi duplikasi kode, mempermudah perawatan, dan sangat cocok untuk pengembangan fitur seperti reservasi online.

Source 2 - SlotParkir (Inheritance)

Pewarisan dari AreaParkir
class AreaParkir {
  String zona;
  AreaParkir(this.zona);
  void infoZona() => print("Zona parkir: $zona");
}

class SlotParkir extends AreaParkir {
  int nomorSlot;
  bool tersedia;
  String? kendaraanPlat;

  SlotParkir(this.nomorSlot, String zona, this.tersedia) : super(zona);

  bool tempatiSlot(String plat) {
    if (tersedia) {
      kendaraanPlat = plat;
      tersedia = false;
      print("✅ Slot $nomorSlot (Zona $zona) ditempati oleh $plat");
      return true;
    } else {
      print("❌ Slot $nomorSlot sudah terisi!");
      return false;
    }
  }

  void bebaskanSlot() {
    if (!tersedia) {
      print("🆓 Slot $nomorSlot dibebaskan.");
      kendaraanPlat = null;
      tersedia = true;
    } else {
      print("Slot sudah kosong.");
    }
  }
}

void main() {
  print("== DEMO SLOT PARKIR + INHERITANCE ==");
  SlotParkir slot1 = SlotParkir(1, "A1", true);
  slot1.infoZona();
  slot1.tempatiSlot("B 1234 AB");
  slot1.bebaskanSlot();
  slot1.tempatiSlot("D 9876 ZZ");
  print("Status akhir tersedia: ${slot1.tersedia}");
}

class TransaksiParkir {
  String id;
  String platKendaraan;
  int durasiJam;
  double tarifAwal = 4000;
  double tarifPerJam = 2000;

  TransaksiParkir(this.id, this.platKendaraan, this.durasiJam);
  
  double hitungTotal() {
    if (durasiJam <= 1) return tarifAwal;
    return tarifAwal + (durasiJam - 1) * tarifPerJam;
  }

  void cetakStruk() {
    double total = hitungTotal();
    print("🧾 STRUK PARKIR");
    print("ID Transaksi: $id");
    print("Kendaraan   : $platKendaraan");
    print("Durasi       : $durasiJam jam");
    print("Total Bayar  : Rp${total.toStringAsFixed(0)}");
    print("-----------------------");
  }
}

class PetugasParkir {
  String nama;
  String shift;
  PetugasParkir(this.nama, this.shift);
  
  void laporan(int slotTerisi, int totalSlot) {
    print("👮 $nama (Shift $shift) => Slot terisi $slotTerisi dari $totalSlot");
  }
}

void main() {
  print("== DEMO TRANSAKSI & PETUGAS ==");
  TransaksiParkir trx = TransaksiParkir("TRX99", "B 8888 XYZ", 3);
  trx.cetakStruk();
  
  PetugasParkir petugas = PetugasParkir("Usman Yakin", "Pagi");
  petugas.laporan(4, 10);
  
  PetugasParkir petugas2 = PetugasParkir("Putra Dhafir", "Siang");
  petugas2.laporan(7, 10);
}

// ========== SMART PARKING FULL SIMULATION ==========
// Developer: USMAN YAKIN (251410006) & PUTRA DHAFIR (251410009)

class Kendaraan {
  String platNomor;
  String jenis;
  DateTime waktuMasuk;
  Kendaraan(this.platNomor, this.jenis, this.waktuMasuk);
  void info() => print("$jenis [$platNomor] masuk jam ${waktuMasuk.hour}:${waktuMasuk.minute}");
}

class SlotParkir {
  int nomor;
  String zona;
  bool tersedia;
  Kendaraan? kendaraan;
  SlotParkir(this.nomor, this.zona, this.tersedia);
  
  bool tempati(Kendaraan k) {
    if (tersedia) {
      kendaraan = k;
      tersedia = false;
      print("✅ Slot $nomor (Zona $zona) ditempati ${k.platNomor}");
      return true;
    }
    print("❌ Slot $nomor penuh");
    return false;
  }
  
  void kosongkan() {
    if (!tersedia && kendaraan != null) {
      print("🆓 Slot $nomor dikosongkan dari ${kendaraan!.platNomor}");
      kendaraan = null;
      tersedia = true;
    }
  }
}

class Transaksi {
  String idTransaksi;
  Kendaraan kendaraan;
  SlotParkir slot;
  DateTime waktuKeluar;
  double biaya;
  Transaksi(this.idTransaksi, this.kendaraan, this.slot, this.waktuKeluar, this.biaya);
  
  double hitungBiaya(int durasiJam) {
    if (durasiJam <= 1) return 5000;
    return 5000 + (durasiJam - 1) * 2500;
  }
  
  void cetak() {
    print("\n🧾 === STRUK SMART PARKING ===");
    print("ID Transaksi : $idTransaksi");
    print("Kendaraan    : ${kendaraan.platNomor} (${kendaraan.jenis})");
    print("Slot         : ${slot.nomor} - ${slot.zona}");
    print("Keluar       : ${waktuKeluar.hour}:${waktuKeluar.minute}");
    print("Total Bayar  : Rp${biaya.toStringAsFixed(0)}");
    print("===============================\n");
  }
}

class Petugas {
  String nama;
  String shift;
  Petugas(this.nama, this.shift);
  void report(List<SlotParkir> slots) {
    int terisi = slots.where((s) => !s.tersedia).length;
    print("📊 Petugas $nama ($shift) : Terisi $terisi dari ${slots.length} slot.");
  }
}

void main() {
  print("🏢 SISTEM SMART PARKING PREMIUM 🏢\n");
  List<SlotParkir> slots = [
    SlotParkir(1, "A", true), SlotParkir(2, "A", true),
    SlotParkir(3, "B", true), SlotParkir(4, "B", true)
  ];
  
  Petugas petugas = Petugas("Usman Yakin Trio", "Pagi");
  
  DateTime now = DateTime(2026, 6, 10, 8, 0);
  Kendaraan mobil = Kendaraan("B 4567 ETR", "Mobil", now);
  mobil.info();
  SlotParkir? slotAvail = slots.firstWhere((s) => s.tersedia, orElse: () => null as SlotParkir?);
  if (slotAvail != null) slotAvail.tempati(mobil);
  
  Kendaraan motor = Kendaraan("AG 1919 Z", "Motor", DateTime(2026, 6, 10, 9, 0));
  motor.info();
  SlotParkir? slotMotor = slots.firstWhere((s) => s.tersedia, orElse: () => null);
  if (slotMotor != null) slotMotor.tempati(motor);
  
  DateTime keluar = DateTime(2026, 6, 10, 11, 0);
  int durasi = keluar.difference(mobil.waktuMasuk).inHours;
  if (durasi == 0) durasi = 1;
  Transaksi trx = Transaksi("TRX001", mobil, slotAvail!, keluar, 0);
  double bayar = trx.hitungBiaya(durasi);
  trx.biaya = bayar;
  trx.cetak();
  slotAvail.kosongkan();
  
  petugas.report(slots);
  print("\n✅ Simulasi Selesai. Coding OOP siap diintegrasi IoT.");
}
